# 数式の例

インライン数式: $\sigma = \sqrt{\frac{1}{n}\sum_{i=1}^{n}(x_i - \mu)^2}$

ディスプレイ数式:

$$
\Sigma = \sum_{i=1}^{n} x_i
$$

ギリシャ文字: $\alpha, \beta, \gamma, \sigma, \Sigma$

$$
\Sigma^{1} =\sum_{k=1}^{n-1} k^1 = \frac{n^{2}}{2}-\frac{n}{2}
$$

$$
\Sigma^{2} =\sum_{k=1}^{n-1} k^2 = \frac{n^{3}}{3}-\frac{n^{2}}{2}+\frac{n}{6}
$$

$$
\Sigma^{1,1} =\sum_{k=1}^{n-1} (n-k)k = n\sum_{k=1}^{n-1} k - \sum_{k=1}^{n-1} k^2 = \frac{n^3}{6}-\frac{n}{6}
$$

$$
\Sigma^{3} =\sum_{k=1}^{n-1} k^3 = \frac{n^{4}}{4}-\frac{n^{3}}{2}+\frac{n^{2}}{4}
$$

$$
\Sigma^{1,2} =\sum_{k=1}^{n-1} (n-k)k^2 = n\sum_{k=1}^{n-1} k^2 - \sum_{k=1}^{n-1} k^3 = \frac{n^{4}}{12}-\frac{n^{2}}{12}
$$

$$
\Sigma^{4} =\sum_{k=1}^{n-1} k^4 = \frac{n^{5}}{5}-\frac{n^{4}}{2}+\frac{n^{3}}{3}-\frac{n}{30}
$$

$$
\Sigma^{1,3} =\sum_{k=1}^{n-1} (n-k)k^3 = n\sum_{k=1}^{n-1} k^3 - \sum_{k=1}^{n-1} k^4 = \frac{n^{5}}{20}-\frac{n^{3}}{12}+\frac{n}{30}
$$

$$
\Sigma^{2,2} =\sum_{k=1}^{n-1} (n-k)^2k^2 = n^2\sum_{k=1}^{n-1} k^2 - 2n\sum_{k=1}^{n-1} k^3 + \sum_{k=1}^{n-1} k^4 = \frac{n^{5}}{30}-\frac{n}{30} 
$$

$$
\Sigma^{5} =\sum_{k=1}^{n-1} k^5 = \frac{n^{6}}{6}-\frac{n^{5}}{2}+\frac{5n^{4}}{12}-\frac{n^{2}}{12}
$$

$$
\Sigma^{1,4} =\sum_{k=1}^{n-1} (n-k)k^4 = n\sum_{k=1}^{n-1} k^4 - \sum_{k=1}^{n-1} k^5 = \frac{n^{6}}{30}-\frac{n^{4}}{12}+\frac{n^{2}}{20}
$$

$$
\Sigma^{2,3} =\sum_{k=1}^{n-1} (n-k)^2k^3 = n^2\sum_{k=1}^{n-1} k^3 - 2n\sum_{k=1}^{n-1} k^4 + \sum_{k=1}^{n-1} k^5 = n^2\left(\frac{n^{4}}{4}-\frac{n^{3}}{2}+\frac{n^{2}}{4}\right) - 2n\left(\frac{n^{5}}{5}-\frac{n^{4}}{2}+\frac{n^{3}}{3}-\frac{n}{30}\right) + \left(\frac{n^{6}}{6}-\frac{n^{5}}{2}+\frac{5n^{4}}{12}-\frac{n^{2}}{12}\right) = \frac{n^{6}}{60}-\frac{n^{2}}{60}
$$

$$
\Sigma^{6} =\sum_{k=1}^{n-1} k^6 = \frac{n^{7}}{7}-\frac{n^{6}}{2}+\frac{n^{5}}{2}-\frac{n^{3}}{6}+\frac{n}{42}
$$

$$
\Sigma^{1,5} =\sum_{k=1}^{n-1} (n-k)k^5 = n\sum_{k=1}^{n-1} k^5 - \sum_{k=1}^{n-1} k^6 = n\left(\frac{n^{6}}{6}-\frac{n^{5}}{2}+\frac{5n^{4}}{12}-\frac{n^{2}}{12}\right) - \left(\frac{n^{7}}{7}-\frac{n^{6}}{2}+\frac{n^{5}}{2}-\frac{n^{3}}{6}+\frac{n}{42}\right) = \frac{n^{7}}{42}-\frac{n^{5}}{12}+\frac{n^{3}}{12}-\frac{n}{42}
$$

$$
\Sigma^{2,4} =\sum_{k=1}^{n-1} (n-k)^2k^4 = n^2\sum_{k=1}^{n-1} k^4 - 2n\sum_{k=1}^{n-1} k^5 + \sum_{k=1}^{n-1} k^6 = n^2\left(\frac{n^{5}}{5}-\frac{n^{4}}{2}+\frac{n^{3}}{3}-\frac{n}{30}\right) - 2n\left(\frac{n^{6}}{6}-\frac{n^{5}}{2}+\frac{5n^{4}}{12}-\frac{n^{2}}{12}\right) + \left(\frac{n^{7}}{7}-\frac{n^{6}}{2}+\frac{n^{5}}{2}-\frac{n^{3}}{6}+\frac{n}{42}\right) = \frac{n^{7}}{105}-\frac{n^{3}}{30}+\frac{n}{42}
$$

$$
\Sigma^{3,3} =\sum_{k=1}^{n-1} (n-k)^3k^3 = n^3\sum_{k=1}^{n-1} k^3 - 3n^2\sum_{k=1}^{n-1} k^4 + 3n\sum_{k=1}^{n-1} k^5 - \sum_{k=1}^{n-1} k^6 = n^3\left(\frac{n^{4}}{4}-\frac{n^{3}}{2}+\frac{n^{2}}{4}\right) - 3n^2\left(\frac{n^{5}}{5}-\frac{n^{4}}{2}+\frac{n^{3}}{3}-\frac{n}{30}\right) + 3n\left(\frac{n^{6}}{6}-\frac{n^{5}}{2}+\frac{5n^{4}}{12}-\frac{n^{2}}{12}\right) - \left(\frac{n^{7}}{7}-\frac{n^{6}}{2}+\frac{n^{5}}{2}-\frac{n^{3}}{6}+\frac{n}{42}\right) = \frac{n^{7}}{140}+\frac{n^{3}}{60}-\frac{n}{42}
$$

$$
\Sigma^{7} =\sum_{k=1}^{n-1} k^7 = \frac{n^{8}}{8}-\frac{n^{7}}{2}+\frac{7n^{6}}{12}-\frac{7n^{4}}{24}+\frac{n^{2}}{12}
$$

$$
\Sigma^{1,6} =\sum_{k=1}^{n-1} (n-k)k^6 = \frac{n^{8}}{56}-\frac{n^{6}}{12}+\frac{n^4}{8}-\frac{5n^2}{84}
$$

$$
\Sigma^{2,5} =\sum_{k=1}^{n-1} (n-k)^2k^5 = \frac{n^{8}}{168}-\frac{n^{4}}{24}+\frac{n^2}{28}
$$

$$
\Sigma^{3,4} =\sum_{k=1}^{n-1} (n-k)^3k^4 = \frac{n^{8}}{280}+\frac{n^{4}}{120}-\frac{n^2}{84}
$$

$$
\Sigma^{8} =\sum_{k=1}^{n-1} k^8 = \frac{n^{9}}{9}-\frac{n^{8}}{2}+\frac{2n^{7}}{3}-\frac{7n^{5}}{15}+\frac{2n^{3}}{9}-\frac{n}{30}
$$

$$
\Sigma^{1,7} =\sum_{k=1}^{n-1} (n-k)k^7 = \frac{n^{9}}{72}-\frac{n^{7}}{12}+\frac{7n^3}{40}-\frac{5n^3}{36}+\frac{n}{30}
$$

$$
\Sigma^{2,6} =\sum_{k=1}^{n-1} (n-k)^2k^6 = \frac{n^{9}}{252}-\frac{n^{5}}{20}+\frac{5n^3}{36}-\frac{n}{30}
$$

$$
\Sigma^{3,5} =\sum_{k=1}^{n-1} (n-k)^3k^5 = \frac{n^{9}}{504}+\frac{n^{5}}{120}-\frac{11n^3}{252}+\frac{n}{30}
$$

$$
\Sigma^{4,4} =\sum_{k=1}^{n-1} (n-k)^4k^4 = \frac{n^{9}}{630}+\frac{2n^3}{63}-\frac{n}{30}
$$

$$
\Sigma^{9} =\sum_{k=1}^{n-1} k^9 = \frac{n^{10}}{10}-\frac{n^{9}}{2}+\frac{3n^{8}}{4}-\frac{7n^{6}}{10}+\frac{n^{4}}{2}-\frac{3n^{2}}{20}
$$

$$
\Sigma^{1,8} =\sum_{k=1}^{n-1} (n-k)k^7 = \frac{n^{10}}{90}-\frac{n^{8}}{12}+\frac{7n^6}{30}-\frac{5n^4}{18}+\frac{7n^2}{60}
$$

$$
\Sigma^{2,7} =\sum_{k=1}^{n-1} (n-k)^2k^7 = \frac{n^{10}}{360}-\frac{7n^{6}}{120}+\frac{5n^4}{36}-\frac{n^2}{12}
$$

$$
\Sigma^{3,6} =\sum_{k=1}^{n-1} (n-k)^3k^6 = \frac{n^{10}}{840}+\frac{n^{6}}{129}-\frac{5n^4}{84}+\frac{n^2}{20}
$$

$$
\Sigma^{4,5} =\sum_{k=1}^{n-1} (n-k)^4k^5 = \frac{n^{10}}{1260}+\frac{n^4}{63}-\frac{n^2}{60}
$$

$$
\Sigma^{10} =\sum_{k=1}^{n-1} k^{10} = \frac{n^{11}}{11}-\frac{n^{10}}{2}+\frac{5n^{9}}{6}-n^{7}+n^{5}-\frac{n^{3}}{2}+\frac{5n}{66}
$$
