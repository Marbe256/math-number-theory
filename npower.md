$$
n^3 = n + 3!\Sigma^{1,1} = n + \sum_{k=1}^{n-1} (n-k)k 
$$

$$
n^4 = n^2 + \frac{4!}{2}\Sigma^{1,2} = n^2 + \frac{4!}{2}\sum_{k=1}^{n-1} (n-k)k^2 
$$

$$
n^5 = n + \frac{5!}{2!2!}\Sigma^{2,2} = n + \frac{5!}{2!2!}\sum_{k=1}^{n-1} (n-k)^2k^2
$$

$$
n^6 = n^2 + \frac{6!}{2!3!}\Sigma^{2,3} = n^2 + \frac{6!}{2!3!}\sum_{k=1}^{n-1} (n-k)^2k^3
$$

$$
n^7 =  \frac{10n}{3}-\frac{7n^3}{3}+\frac{7!}{3!3!}\Sigma^{3,3} = \frac{n}{3}(10-7n^2)+\frac{7!}{3!3!}\sum_{k=1}^{n-1} (n-k)^3k^3
$$

$$
n^{8} =  \frac{10 n^{2}}{3} - \frac{7 n^{4}}{3}  + \frac{8!}{3!4!}\Sigma^{3,4} = \frac{n^{2}}{3}(10 - 7n^2) + \frac{8!}{3!4!}\sum_{k=1}^{n-1} (n-k)^{3}k^{4}
$$

$$
n^{9} = 21 n - 20 n^{3}  + \frac{9!}{4!4!}\Sigma^{4,4} = n(21 - 20 n^2) + \frac{9!}{4!4!}\sum_{k=1}^{n-1} (n-k)^{4}k^{4}
$$

$$
n^{10} =  21 n^{2} - 20 n^{4}  + \frac{10!}{4!5!}\Sigma^{4,5} = n^2( 21 - 20 n^{2} ) + \frac{10!}{4!5!}\sum_{k=1}^{n-1} (n-k)^{4}k^{5}
$$

$$
n^{11} = 210 n - 231 n^{3} + 22 n^{5} + \frac{11!}{5!5!}\Sigma^{5,5} = n( 210 - 231 n^{2} + 22 n^{4} ) + \frac{11!}{5!5!}\sum_{k=1}^{n-1} (n-k)^{5}k^{5}
$$

$$
n^{12} = 210 n^{2} - 231 n^{4} + 22 n^{6}  \frac{12!}{5!6!}\Sigma^{5,6} =n^2(210 - 231 n^{2} + 22 n^{4} ) + \frac{12!}{5!6!}\sum_{k=1}^{n-1} (n-k)^{5}k^{6}
$$
